## cut before 'Game start'
data = data.split('Game start')[-1]
# print(data)


def get_valut(data, target_list, target_word):
    temp_data = data.split(target_word)
    for temp_step in temp_data[1:]:
        target_value = temp_step.split(' ')[0]
        if target_value[-1] == ',':
            target_value = target_value[:-1]
        target_value = float(target_value)
        target_list.append(target_value)
    return target_list


step_list = get_valut(data, step_list, step_word)
length_list = get_valut(data, length_list, length_word)
area_list = get_valut(data, area_list, area_word)
reward_list = get_valut(data, reward_list, reward_word)
